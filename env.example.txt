# Supabase Configuration (Example)
# Copy these values to your Vercel project's environment variables

NEXT_PUBLIC_SUPABASE_URL=https://your-project.supabase.co
NEXT_PUBLIC_SUPABASE_ANON_KEY=your-anon-key-here
SUPABASE_SERVICE_ROLE_KEY=your-service-role-key-here

# Optional: For development
NEXT_PUBLIC_APP_URL=http://localhost:3000

# Instructions:
# 1. Get these values from your Supabase Dashboard → Settings → API
# 2. Add them to Vercel Dashboard → Project Settings → Environment Variables
# 3. For local development, create a .env.local file with these values
